<script setup>
import { ref, reactive, watch, computed, onMounted, onUnmounted } from 'vue';
import Sortable from 'sortablejs';

const props = defineProps({ 
  cards: Array,
  selectedCards: Array,
  categoryId: Number,
  subCategoryId: [Number, null],
  isMultiSelectMode: Boolean
});

const emit = defineEmits([
  'cardsReordered', 
  'contextEdit', 
  'contextDelete',
  'toggleCardSelection',
  'openMovePanel',
  'cardMoved'
]);

const cardGridRef = ref(null);
const localCards = ref([]);
let sortableInstance = null;
const isDragging = ref(false);

const contextMenuVisible = ref(false);
const contextMenuX = ref(0);
const contextMenuY = ref(0);
const contextMenuCard = ref(null);

let preventClick = false;

const isMultiSelectMode = computed(() => props.isMultiSelectMode);
const shouldDisableLink = computed(() => isDragging.value || isMultiSelectMode.value);

watch(() => props.cards, (newCards) => {
  if (newCards) {
    localCards.value = [...newCards];
  }
}, { immediate: true, deep: true });

function initSortable() {
  if (!cardGridRef.value) return;
  
  if (isMultiSelectMode.value) {
    if (sortableInstance) {
      sortableInstance.destroy();
      sortableInstance = null;
    }
    return;
  }

  if (sortableInstance) sortableInstance.destroy();

  sortableInstance = new Sortable(cardGridRef.value, {
    group: {
      name: 'shared-cards',
      pull: true,
      put: true
    },
    animation: 200,
    delay: 200,
    delayOnTouchOnly: true,
    ghostClass: 'sortable-ghost',
    dragClass: 'sortable-drag',
    chosenClass: 'sortable-chosen',
    forceFallback: false,
    onStart: (evt) => {
      isDragging.value = true;
      preventClick = true;
      document.body.classList.add('dragging-active');
    },
    onEnd: (evt) => {
      isDragging.value = false;
      document.body.classList.remove('dragging-active');
      
      setTimeout(() => {
        preventClick = false;
      }, 100);

      const { newIndex, oldIndex, from, to } = evt;
      
      if (!to.isEqualNode(from)) {
        return;
      }

      if (newIndex === oldIndex) {
        return;
      }

      const item = localCards.value.splice(oldIndex, 1)[0];
      localCards.value.splice(newIndex, 0, item);

      const cardIds = localCards.value.map(c => c.id);
      emit('cardsReordered', cardIds, props.categoryId, props.subCategoryId);
    },
    onAdd: (evt) => {
        const cardId = parseInt(evt.item.dataset.cardId);
        const newIndex = evt.newIndex;
        const rawFromSubId = evt.from.dataset.subCategoryId;
        const fromSubCategoryId = (rawFromSubId && rawFromSubId !== 'null' && rawFromSubId !== 'undefined') 
          ? parseInt(rawFromSubId) 
          : null;
        
        emit('cardMoved', {
          cardId,
          newIndex,
          fromSubCategoryId,
          toSubCategoryId: props.subCategoryId,
          toCategoryId: props.categoryId
        });
      },
    onRemove: (evt) => {
      const cardId = parseInt(evt.item.dataset.cardId);
      localCards.value = localCards.value.filter(c => c.id !== cardId);
    }
  });
}

watch(isMultiSelectMode, () => {
  initSortable();
});

onMounted(() => {
  initSortable();
  document.addEventListener('click', handleClickOutside);
  document.addEventListener('scroll', closeContextMenu);
});

onUnmounted(() => {
  if (sortableInstance) sortableInstance.destroy();
  document.removeEventListener('click', handleClickOutside);
  document.removeEventListener('scroll', closeContextMenu);
  document.body.classList.remove('dragging-active');
});

function handleContextMenu(event, card) {
  if (isDragging.value) return;
  
  contextMenuCard.value = card;
  
  const menuWidth = 180;
  const menuHeight = 280;
  const viewportWidth = window.innerWidth;
  const viewportHeight = window.innerHeight;
  const padding = 10;
  
  let x = event.clientX;
  let y = event.clientY;
  
  if (x + menuWidth + padding > viewportWidth) {
    x = viewportWidth - menuWidth - padding;
  }
  if (x < padding) {
    x = padding;
  }
  
  if (y + menuHeight + padding > viewportHeight) {
    y = viewportHeight - menuHeight - padding;
  }
  if (y < padding) {
    y = padding;
  }
  
  contextMenuX.value = x;
  contextMenuY.value = y;
  contextMenuVisible.value = true;
}

function showCardMenu(event, card) {
  if (isDragging.value) return;
  
  contextMenuCard.value = card;
  
  const menuWidth = 180;
  const menuHeight = 280;
  const viewportWidth = window.innerWidth;
  const viewportHeight = window.innerHeight;
  const padding = 10;
  
  const rect = event.currentTarget.getBoundingClientRect();
  let x = rect.right - menuWidth;
  let y = rect.bottom + 4;
  
  if (x < padding) x = padding;
  if (x + menuWidth + padding > viewportWidth) {
    x = viewportWidth - menuWidth - padding;
  }
  if (y + menuHeight + padding > viewportHeight) {
    y = rect.top - menuHeight - 4;
  }
  if (y < padding) y = padding;
  
  contextMenuX.value = x;
  contextMenuY.value = y;
  contextMenuVisible.value = true;
}

function closeContextMenu() {
  contextMenuVisible.value = false;
  contextMenuCard.value = null;
}

function onContextEdit() {
  if (contextMenuCard.value) {
    emit('contextEdit', contextMenuCard.value);
  }
  closeContextMenu();
}

function onContextDelete() {
  if (contextMenuCard.value) {
    emit('contextDelete', contextMenuCard.value);
  }
  closeContextMenu();
}

function onContextSelect() {
  if (contextMenuCard.value) {
    emit('toggleCardSelection', contextMenuCard.value);
  }
  closeContextMenu();
}

function onContextMove() {
  if (contextMenuCard.value) {
    if (!isCardSelected(contextMenuCard.value)) {
      emit('toggleCardSelection', contextMenuCard.value);
    }
    emit('openMovePanel');
  }
  closeContextMenu();
}

function onContextOpen() {
  if (contextMenuCard.value) {
    window.open(contextMenuCard.value.url, '_blank');
  }
  closeContextMenu();
}

async function onContextCopyUrl() {
  if (contextMenuCard.value) {
    const url = contextMenuCard.value.url;
    try {
      await navigator.clipboard.writeText(url);
    } catch (err) {
      const textarea = document.createElement('textarea');
      textarea.value = url;
      textarea.style.position = 'fixed';
      textarea.style.opacity = '0';
      document.body.appendChild(textarea);
      textarea.select();
      document.execCommand('copy');
      document.body.removeChild(textarea);
    }
  }
  closeContextMenu();
}

function handleClickOutside(event) {
  if (contextMenuVisible.value) {
    closeContextMenu();
  }
}

function handleLinkClick(event, card) {
  if (preventClick || isDragging.value) {
    event.preventDefault();
    event.stopPropagation();
    return;
  }
  
  if (event.ctrlKey || event.metaKey) {
    event.preventDefault();
    event.stopPropagation();
    emit('toggleCardSelection', card);
    return;
  }
  
  if (props.isMultiSelectMode) {
    event.preventDefault();
    event.stopPropagation();
    emit('toggleCardSelection', card);
    return;
  }
}

function handleMouseUp(event, card) {
  if (preventClick) return;
  
  if (props.isMultiSelectMode && !isDragging.value) {
    event.preventDefault();
    event.stopPropagation();
    emit('toggleCardSelection', card);
  }
}

function handleMouseDown(event, card) {
  if (event.button !== 0) return;
  
  if (event.ctrlKey || event.metaKey) {
    event.preventDefault();
    event.stopPropagation();
    emit('toggleCardSelection', card);
    preventClick = true;
  }
}

function getOriginUrl(url) {
  try {
    const urlObj = new URL(url);
    return urlObj.origin;
  } catch {
    return null;
  }
}

function getLogo(card) {
  if (!card) return '/default-favicon.png';
  
  if (card.logo_url) {
    return card.logo_url;
  }
  
  const originUrl = getOriginUrl(card.url);
  if (originUrl) {
    return `https://api.xinac.net/icon/?url=${originUrl}&sz=128`;
  }
  
  return '/default-favicon.png';
}

const CDN_PROVIDERS = [
  (url) => `https://api.xinac.net/icon/?url=${url}&sz=128`,
  (url) => `https://api.afmax.cn/so/ico/index.php?r=${url}&sz=128`,
  (url) => `https://icon.horse/icon/${url}`,
  (url) => `https://www.google.com/s2/favicons?domain=${url}&sz=128`,
  (url) => `https://favicon.im/${url}?larger=true`,
];

function onImgError(e, card) {
  const currentSrc = e.target.src;
  const originUrl = getOriginUrl(card.url);
  
  if (!originUrl) {
    e.target.src = '/default-favicon.png';
    return;
  }
  
  if (e.target._cdnIndex === undefined) e.target._cdnIndex = 0;
  
  for (let i = 0; i < CDN_PROVIDERS.length; i++) {
    const cdnUrl = CDN_PROVIDERS[i](originUrl);
    if (currentSrc.includes('api.xinac.net') && i === 0 ||
        currentSrc.includes('api.afmax.cn') && i === 1 ||
        currentSrc.includes('icon.horse') && i === 2 ||
        currentSrc.includes('www.google.com/s2/favicons') && i === 3 ||
        currentSrc.includes('favicon.im') && i === 4) {
      if (i + 1 < CDN_PROVIDERS.length) {
        e.target._cdnIndex = i + 1;
        e.target.src = CDN_PROVIDERS[i + 1](originUrl);
        return;
      }
      break;
    }
  }
  
  e.target.src = '/default-favicon.png';
}

function getTooltip(card) {
  if (!card) return '';
  let tip = '';
  if (card.desc) tip += card.desc + '\n';
  if (card.tags && card.tags.length > 0) {
    tip += 'Ê†áÁ≠æ: ' + card.tags.map(t => t.name).join(', ') + '\n';
  }
  tip += card.url;
  return tip;
}

function truncate(str) {
  if (!str) return '';
  return str.length > 20 ? str.slice(0, 20) + '...' : str;
}

function isCardSelected(card) {
  return props.selectedCards?.some(c => c.id === card.id) || false;
}
</script>
